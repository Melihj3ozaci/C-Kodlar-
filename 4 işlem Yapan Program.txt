using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace ConsoleApplication4
{
    class Program
    {

        {

            double toplama, cikartma, carpma, bolme;
            Console.Write("1.Sayýyý Giriniz = ");

            Console.Write("2 Sayýyý Giriniz = ");
            y = Convert.ToInt32(Console.ReadLine());

            cikartma = x - y;
            bolme = x / y;
            carpma = x * y;
            Console.WriteLine(x + " + " + y +" = " + toplama);
            Console.WriteLine(x + " - " + y + " = " + +cikartma);
            Console.WriteLine(x + " * " + y + " = " + +bolme);
            Console.WriteLine(x + " / " + y + " = " + +carpma);
            Console.Write("Programdan Çýkmak Ýçin Enter'e Basýnýz");

        }
